## Examples

### AvatarLink using non-existing image with fallback to initials

```js live
<AvatarLink src="" href="https://traefik.io">
  KS
</AvatarLink>
```

### AvatarLink using non-existing image with fallback to custom initials

```js live
<AvatarLink src="" href="https://traefik.io">
  <Text sx={{ color: 'white' }} marginBottom={1}>
    K
  </Text>
  <Text sx={{ color: 'white' }} marginTop={1}>
    S
  </Text>
</AvatarLink>
```

### AvatarLink using existing image on an alternative variant

```js live removeFragment
function AlternativeVariant() {
  const darkMode = useDarkMode()
  const isDarkMode = darkMode.value

  return (
    <Box sx={{ bg: 'dark' }} p={2}>
      <AvatarLink
        src={getValidImageUrl()}
        sx={{ bg: 'purple' }}
        href="https://traefik.io"
        variant={isDarkMode ? 'light' : 'dark'}
      >
        KS
      </AvatarLink>
    </Box>
  )
}
```
